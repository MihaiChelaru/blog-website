Bar graph

## Calculate descriptive statistics for each group
descriptive_stats = r.HR.describe()   
mean_HR = descriptive_stats.iloc[1].values   
ste_HR = descriptive_stats.iloc[2]/descriptive_stats.iloc[0]    
 
## Plot bar graph with error bars
plt.figure(1)
x = r.HR.columns
y = mean_HR
x_pos = [i for i, _ in enumerate(x)]
plt.bar(x_pos, y, yerr=ste_HR, color='green')
plt.xticks(x_pos, x)
